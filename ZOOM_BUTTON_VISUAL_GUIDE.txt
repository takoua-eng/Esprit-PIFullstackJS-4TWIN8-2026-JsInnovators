📝 UTILISATION DANS LE CODE
═══════════════════════════════════════════════════════════════════════════

// 1. Importer le service
import { ZoomService } from 'src/app/services/zoom.service';

// 2. L'injecter dans le constructeur
export class MyComponent {
  constructor(private zoom: ZoomService) {}
  
  // 3. Utiliser
  ngOnInit() {
    // Obtenir le zoom
    const nivel = this.zoom.getCurrentZoom();          // 100, 125, etc.
    const percent = this.zoom.getCurrentZoomPercentage();  // "150%"
    const ratio = this.zoom.getZoomRatio();            // 1.5
    
    // Modifier le zoom
    this.zoom.setZoom(150);
    this.zoom.zoomIn();      // +25%
    this.zoom.zoomOut();     // -25%
    this.zoom.resetZoom();   // → 100%
    
    // Vérifier les limites
    if (this.zoom.canZoomIn()) { ... }
    if (this.zoom.canZoomOut()) { ... }
    
    // Écouter les changements
    window.addEventListener('zoom-changed', (event: CustomEvent) => {
      console.log('Zoom level:', event.detail.level);
      console.log('Ratio:', event.detail.ratio);
    });
  }
}
